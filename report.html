from sklearn.tree import DecisionTreeRegressor
def performance_metric(y_true, y_predict):
    """"""
    y_true = np.array(y_true)
    y_predict = np.array(y_predict)
    mean_value = np.mean(y_true)
    ss_tot = np.sum(np.square(y_true-mean_value))
    ss_res = np.sum(np.square(y_true-y_predict))
    score = 1-ss_res/ss_tot

    return score

def cross_val_score(estimator, X, y, scoring = performance_metric,cv=3):
    scores = [0 for x in range(0,cv)]
    # 3
    X_split = np.array_split(X,cv)
    y_split = np.array_split(y,cv)
    for i in range(0,cv):
        X_split_copy = list(X_split)
        y_split_copy = list(y_split)
        X_split_valid = X_split_copy[i]
        y_split_valid = y_split_copy[i]
        
        X_split_copy.pop(i)
        y_split_copy.pop(i)
        X_split_train = pd.DataFrame()
        y_split_train = pd.Series()
        for j in X_split_copy:
            X_split_train = X_split_train.append(j)
        for k in y_split_copy:
            y_split_train = y_split_train.append(k)
            
        estimator.fit(X_split_train,y_split_train)
        score = scoring(y_split_valid,estimator.predict(X_split_valid))
        scores.append(score)
    
    return scores
def fit_model2(X, y):
    """  [X,y]"""
    max_depth = range(1,11)
    #
    best_estimator = None
    best_score = 0
    for depth in max_depth:
        estimator = DecisionTreeRegressor(max_depth=depth)
        scores = cross_val_score(estimator,X,y)
        score = np.mean(scores)
        if score > best_score:
            best_score = score
            best_estimator = estimator
    return best_estimator

from sklearn.tree import DecisionTreeRegressor
def performance_metric(y_true, y_predict):
    """"""
    y_true = np.array(y_true)
    y_predict = np.array(y_predict)
    mean_value = np.mean(y_true)
    ss_tot = np.sum(np.square(y_true-mean_value))
    ss_res = np.sum(np.square(y_true-y_predict))
    score = 1-ss_res/ss_tot

    return score

def cross_val_score(estimator, X, y, scoring = performance_metric,cv=3):
    scores = [0,0,0]
    # 3
    X_split = np.array_split(X,3)
    y_split = np.array_split(y,3)
    X_valid1 = X_split[0]
    y_valid1 = y_split[0]
    X_train1 = X_split[1].append(X_split[2])
    y_train1 = y_split[1].append(y_split[2])
    estimator.fit(X_train1,y_train1)
    score1 = scoring(y_valid1,estimator.predict(X_valid1))
    scores.append(score1)
    
    X_valid2 = X_split[1]
    y_valid2 = y_split[1]
    X_train2 = X_split[0].append(X_split[2])
    y_train2 = y_split[0].append(y_split[2])
    estimator.fit(X_train2,y_train2)

    return scores

def fit_model2(X, y):
    """  [X,y]"""
    max_depth = range(1,11)
    #
    best_estimator = None
    best_score = 0
    for depth in max_depth:
        estimator = DecisionTreeRegressor(max_depth=depth)
        scores = cross_val_score(estimator,X,y)
        score = np.mean(scores)
        if score > best_score:
            best_score = score
            best_estimator = estimator
    return best_estimator

    """  [X,y]"""
    
    cross_validator = KFold()

    grid = grid.fit(X, y)

    return grid.best_estimator_
